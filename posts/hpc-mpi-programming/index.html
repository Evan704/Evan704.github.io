#include<iostream>
#include"mpi.h"
#include<chrono>
double f(double x) {
    return x*x;
}
double inte(double a, int num, double width) {
    double ans = 0;
    for(int i = 0; i < num; i++) {
        ans += (f(a+i*width)+f(a+(i+1)*width))*width/2.0;
    }
    return ans;
}
int main(int argc, char *argv[]) {
    double a = 1.0, b = 4.0;
    int rank, size, n = 600000;
    double width = (b-a)/n;
    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    auto start = std::chrono::steady_clock::now();
    int num = n/size;
    double la = a+num*rank*width;
    double lans = inte(la, num, width);
    if(rank != 0) {
        MPI_Send(&lans, 1, MPI_DOUBLE, 0, 0, MPI_COMM_WORLD);
    }
    else {
        double ans = lans;
        for(int i = 1; i < size; i++) {
            MPI_Recv(&lans, 1, MPI_DOUBLE, i, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
            ans += lans;
        }
        auto end = std::chrono::steady_clock::now();
        auto dura = std::chrono::duration_cast<std::chrono::microseconds>(end-start);
        std::cout << ans << std::endl;
        std::cout << dura.count() << std::endl; // 1347
        start = std::chrono::steady_clock::now();
        std::cout << inte(a, n, width) << std::endl;
        end = std::chrono::steady_clock::now();
        dura = std::chrono::duration_cast<std::chrono::microseconds>(end-start);
        std::cout << dura.count() << std::endl; // 2669
    }
    MPI_Finalize();
}

此处效率的提高没达到理论估计是因为在不同进程间传输信息也需要时间，而这个任务的计算量太小以至于通信操作显得昂贵。

#include<iostream>
#include"mpi.h"
#include<chrono>
double f(double x) {
    return x*x;
}
double inte(double a, int num, double width) {
    double ans = 0;
    for(int i = 0; i < num; i++) {
        ans += (f(a+i*width)+f(a+(i+1)*width))*width/2.0;
    }
    return ans;
}
int main(int argc, char *argv[]) {
    double a = 1.0, b = 4.0;
    int rank, size, n = 600000;
    double width = (b-a)/n;
    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    auto start = std::chrono::steady_clock::now();
    int num = n/size;
    double la = a+num*rank*width;
    double lans = inte(la, num, width), ans = 0;
    MPI_Reduce(&lans, &ans, 1, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
    if(rank == 0) {
        auto end = std::chrono::steady_clock::now();
        auto dura = std::chrono::duration_cast<std::chrono::microseconds>(end-start);
        std::cout << ans << std::endl;
        std::cout << dura.count() << std::endl;
        start = std::chrono::steady_clock::now();
        std::cout << inte(a, n, width) << std::endl;
        end = std::chrono::steady_clock::now();
        dura = std::chrono::duration_cast<std::chrono::microseconds>(end-start);
        std::cout << dura.count() << std::endl;
    }
    MPI_Finalize();
}
